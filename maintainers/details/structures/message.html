These are the global data parmaeeters associated with the print utilitiy functions.

Globals:

int dds_error;            /* last non-zero errno */

int dds_warn_count;       /* warning counter */
int dds_error_count;      /* error counter */
int dds_debug_count;      /* debug counter */

DDS_STRING dds_warn_msg;  /* last warning message */
DDS_STRING dds_error_msg; /* last error   message */
DDS_STRING dds_debug_msg; /* last debug   message */

FILE* dds_print_fp;       /* print "C" file pointer */
int dds_print_lu;         /* print fortran logical unit */
