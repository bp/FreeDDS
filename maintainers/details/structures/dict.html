These are the data structures used in the dictionary utiltiy declarations.

Sturctures:

DDS_LOAD  - Dictionary load state
DDS_DICT  - Dictionary descriptor (one per dictionary)
DDS_DEFN  - Definition hash bucket (extended hash table entry)
DDS_ALIAS - Alias descriptor for definition name

Globals:

DDS_TABLE dds_dict;         /* Dictionary hash table */

DDS_DICT *dds_scan_first;   /* Newest scan dict (NULL == none) */
DDS_DICT *dds_scan_last;    /* Oldest scan dict (NULL == none) */

DDS_DICT *dds_scan_dict;    /* Current scan dict descriptor (NULL == none) */
DDS_DEFN *dds_scan_defn;    /* Current scan defn descriptor */
char *dds_scan_name;        /* Current scan defn name */
size_t dds_scan_value_bias; /* Current scan defn value bias */
DDS_TOKEN dds_scan_token;   /* Current scan token value and next char* */
char dds_scan_value_save;   /* Current scan defn value save char */
DDS_DEFN *dds_scan_newer;   /* Current scan defn newer defn descriptor */

DDS_DICT *dds_print_first;  /* Current newest print dict (NULL == none) */
DDS_DICT *dds_print_last;   /* Current oldest print dict (NULL == none) */

DDS_DICT *dds_log_dict;     /* log dictionary descriptor (NULL == none) */

DDS_ALIAS *dds_alias_free;  /* Malloc alias list (NULL == none) */
DDS_ALIAS *dds_alias_done;  /* Unused alias list (NULL == none) */
unsigned dds_alias_count;   /* Alias count (malloc total) */
