fddz_getdict - Get the name of a dictionary when only the BIN_TAG is known

SYNOPSIS

#include "fdds.h"
#include "fddz.h"

integer function fddz_getdict(dict, bin_tag)

character dict*(*)
integer bin_tag

DESCRIPTION

Function "fddz_getdict" finds the name of a dictionary given it's binary TAG. This is useful when the name is needed after a file was openned with fddx_in or fddx_out functions.

EXAMPLE

integer fd_in, fd_out, fd_buf_out, ier
character buf_in_dict*(*), buf_out_dict*(*), Title*(*)
 ...
c
c Use convenience routines on input but not on output
c
fd_in = fddx_in ('in','stdin:',Title)
ier = fddz_getdict(buf_in_dict, fd_in)
ier = fdds_out(buf_out_dict,' ',' ',buf_in_dict)
fd_buf_out = fdds_open(buf_out_dict,' ',' ','m')
ier = fdds_out(out_dict,' ',' ',buf_out_dict)
fd_out = fdds_open(out_dict,' ',' ','w')
