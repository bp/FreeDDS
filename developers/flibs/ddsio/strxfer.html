strxfer - Broadcasts a string to all nodes in this module's MPI communicator

SYNOPSIS

use MyMPI_mod

subroutine strxfer(string)

character(len=*), intent(inout) :: string

DESCRIPTION

"strxfer" is an MPI collective routine to broadcast a string from node 0 to all other nodes in this module's MPI communicator.

EXAMPLE

	program test
	
	    use MyMPI_mod
	    use dds, only : fdds_prtcon
	
	    integer :: ier
	    character(len=99) :: string
	
	    call initMyMPI()
	
	    if (myid == 0) string='test string'
	    call strxfer(string)
	
	    call finalMyMPI()
	
	end program test
