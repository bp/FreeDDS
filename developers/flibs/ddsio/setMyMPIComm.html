setMyMPIComm - Sets up this module with a given MPI communicator

SYNOPSIS

use MyMPI_mod

subroutine setMyMPIComm(mpicomm)

integer, intent(in) :: mpicomm

DESCRIPTION

"setMyMPIComm" changes this module's MPI communicator from the default (MPI_COMM_WORLD) to a given communicator.

EXAMPLE

	program test
	
	    use MyMPI_mod
	    
	    call initMyMPI()
	    call setMyMPIComm(MPI_COMM_SELF)
	    
	    call finalMyMPI()
	
	end program test
