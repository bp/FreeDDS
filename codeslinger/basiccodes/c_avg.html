/***********************************************************************
 *
 * Get the average value of a dataset
 *
 ***********************************************************************/

#include <stdlib.h>
#include <stdio.h>
#include <cdds.h>

/* main */
int main(int argc, char **argv)
{

/* For DDS functions. Arguments and return values */
   char  *prog, *Title;
   int   ier;
   BIN_TAG fd_in;

/* Data attributes */
   int   n1,n2,n3;

/* Internal dynamic array */
   float *trace;

/* Internal scalar values and loop counters */
   int    i1,i2,i3;
   float  data_avg;
   double data_sum, denom;

/*-------------------------------------------------------------------*/

/* Initialize */
   setargcv(argc, argv);
   prog="c_avg";
   Title="c_avg: Get the average value of a dataset";

/* Open input file */
   fd_in=cddx_in("in","stdin:",Title);

/* Retrieve data characteristics */
   ier=cdds_scanf("size.axis(1)","%d",&n1);
   ier=cdds_scanf("size.axis(2)","%d",&n2);
   n3=cdds_axis_prod(3);

/* Allocate trace array */
   trace=(float*)malloc(n1*sizeof(float));

/* Initialize the sum */
   data_sum=0.0;

/* Loop over input data, accumulating the sum */
   for (i3=0; i3<n3; i3++) {
      for (i2=0; i2<n2; i2++) {
         ier=cddx_read(fd_in,trace,1);
         for (i1=0; i1<n1; i1++) {
            data_sum = data_sum + trace[i1];
         }
      }
   }

/* Compute the average */
   denom = n1*n2*n3;
   data_avg = data_sum/denom;

/* Report the result */
   printf("The average of input samples is %g\n",data_avg);

/* Clean up */
   free(trace);
   ier=cdds_close(fd_in);

/* Exit */
   exit(0);
}
