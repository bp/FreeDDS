!***********************************************************************
!
! transpose subsets of 2D arrays
!
!***********************************************************************

      program f90_transp

      implicit none

#include <f90dds.h>

! Define F90 precision for integer*8 variables
      integer,parameter :: LI=selected_int_kind(18)

! For DDS functions. Arguments and return values
      character(LEN=10),parameter :: prog="f90_transp"
      character(LEN=80)           :: Title
      integer ier, fd_in, fd_out

      integer(KIND=LI),parameter :: izero8=0
      integer(KIND=LI)           :: ier8

! Data attributes
      character(LEN=AXISNAME_MAX) :: axes(RANK_MAX)
      integer n1,n2,n3, rank

! User parameters
      integer i1s,i1e, i2s,i2e
      logical lverb

! Internal dynamic array
      real,allocatable :: record_in(:,:), record_out(:,:)

! Internal loop counters and other scalar values
      integer i3, irank, n1_out,n2_out

!-------------------------------------------------------------------
!
! Initialize the Title
!
      Title=prog//': transpose subsets of 2D arrays'

!
! Open the print file and dump command-line help is requested
!
      ier=fdds_openpr(prog,'')
      if (ier.gt.0) call help(prog,Title)

!
! Open input file
!
      fd_in=fddx_in('in','stdin:',Title)

!
! Retrieve data characteristics
!
      ier=fdds_scanf('size.axis(1)','%d\0',n1)
      ier=fdds_scanf('size.axis(2)','%d\0',n2)
      n3=fdds_axis_prod(3)

      rank=fdds_scank('axis','')
      do irank=1,rank
         ier=fdds_scanf('','%s\0',axes(irank))
      enddo

!
! Retrieve command line and parfile parameters
!
      ier=fdds_dict('par:','scan')

      i1s=1
      i1e=n1
      i2s=1
      i2e=n2
      ier=fdds_scanf('1s','%d\0',i1s)
      ier=fdds_scanf('1e','%d\0',i1e)
      ier=fdds_scanf('2s','%d\0',i2s)
      ier=fdds_scanf('2e','%d\0',i2e)

      lverb=.false.
      if (fdds_scank('verbose',DDS_TRUE_KEY).gt.0) lverb=.true.

!
! Set up the output dictionary
!
      fd_out=fddx_out('out','stdout:',Title,fd_in)

!
! Print modified axis info to output dictionary
!
      ier=fdds_printf('axis',' %s\0',axes(2))
      ier=fdds_printf('',' %s\0',axes(1))
      do irank=3,rank
         ier=fdds_printf('',' %s\0',axes(irank))
      enddo
      ier=fdds_printf('','\n\0')

!
! Twiddle with the dictionary to get axis names to register
!
      ier=fddx_dict(fd_out,'scan')
      ier=fddx_dict(fd_out,'print')

!
! Print modified size info to output dictionary
!
      n1_out=i2e-i2s+1
      n2_out=i1e-i1s+1

      ier=fdds_printf('size.axis(1)','%d\n\0',n1_out)
      ier=fdds_printf('size.axis(2)','%d\n\0',n2_out)

!
! Now actually open the output with a seek
!
      ier8=fdds_lseek8(fd_out,0,izero8,SEEK_SET)

!
! Allocate data arrays
!
      allocate( record_in(n1,n2) )
      allocate( record_out(n1_out,n2_out) )

!
! Loop over n3, read-transpose-write
!
      do i3 = 1,n3

         if (lverb .and. mod(i3,10).eq.0) then
            ier=fdds_prtcon('record %5d of %5d\n\0',i3,n3)
         endif

         if (fddx_read(fd_in,record_in,n2).ne.n2) then
            ier=fdds_prterr('Reading record %d\n\0',i3)
            exit
         endif

         record_out(:,:) = transpose(record_in(i1s:i1e,i2s:i2e))

         if (fddx_write(fd_out,record_out,n2_out).ne.n2_out) then
            ier=fdds_prterr('Writing record %d\n\0',i3)
            exit
         endif

      enddo

!
! Clean up
!
      deallocate(record_in)
      deallocate(record_out)
      ier=fdds_close(fd_in)
      ier=fdds_close(fd_out)
      ier=fdds_closepr()

!
! Stop
!
      stop
      end




!***********************************************************************
!
! subroutine to print help when requested on the command line
!
!***********************************************************************

      subroutine help(prog,Title)

      implicit none

      character(LEN=*) :: prog, Title

      write(0,*)''
      write(0,*)trim(Title)
      write(0,*)''
      write(0,*)' Parameter....Meaning..........................Default'
      write(0,*)''
      write(0,*)'  in=         input dictionary                  stdin:'
      write(0,*)'  out=        output dictionary                stdout:'
      write(0,*)''
      write(0,*)'  1s=         First sample of input to transpose     1'
      write(0,*)'  1e=         Last    "    "    "   "      "        n1'
      write(0,*)'  2s=         First trace  "    "   "      "         1'
      write(0,*)'  2e=         Last    "    "    "   "      "        n2'
      write(0,*)''
      write(0,*)'  verbose=    verbose output                        no'
      write(0,*)''
      write(0,*)'  USAGE: '//trim(prog)
      write(0,*)'           in=[in_dict] out=[out_dict]'
      write(0,*)'           [ 1s=[i1s] 1e=[i1e] 2s=[i2s] 2e=[i2e]'
      write(0,*)'             verbose=[y|n] ]'
      write(0,*)''

      stop

      return
      end
