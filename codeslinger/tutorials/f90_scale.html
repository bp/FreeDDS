!***********************************************************************
!
! multiply data by a scalar, add a bias
!
!***********************************************************************

      program f90_scale

      implicit none

#include <f90dds.h>

! Define F90 precision for integer*8 variables
      integer,parameter :: LI=selected_int_kind(18)

! For DDS functions. Arguments and return values
      character(LEN=9),parameter  :: prog="f90_scale"
      character(LEN=80)           :: Title
      integer ier, fd_in, fd_out

      integer(KIND=LI),parameter :: izero8=0
      integer(KIND=LI)           :: ier8

! Data attributes
      integer n1,n2,n3

! User parameters
      real    scalar, bias

! Internal dynamic array
      real,allocatable :: record(:,:)

! Internal loop counters
      integer i1,i2,i3

!-------------------------------------------------------------------
!
! Initialize the Title
!
      Title=prog//': data_out= scalar*data_in + bias'

!
! Open input file
!
      fd_in=fddx_in('in','stdin:',Title)

!
! Retrieve data characteristics
!
      ier=fdds_scanf('size.axis(1)','%d\0',n1)
      ier=fdds_scanf('size.axis(2)','%d\0',n2)
      n3=fdds_axis_prod(3)

!
! Retrieve command line and parfile parameters
!
      ier=fdds_dict('par:','scan')

      scalar=1.0
      ier=fdds_scanf('scalar s','%f\0',scalar)

      bias=0.0
      ier=fdds_scanf('bias b','%f\0',bias)

!
! Open the output
!
      fd_out=fddx_out('out','stdout:',Title,fd_in)
      ier8=fdds_lseek8(fd_out,0,izero8,SEEK_SET)

!
! Allocate data array
!
      allocate( record(n1,n2) )

!
! Loop over n3, read-scale-write
!
      do i3 = 1,n3
         ier=fddx_read(fd_in,record,n2)
         record(:,:) = scalar*record(:,:) + bias
         ier=fddx_write(fd_out,record,n2)
      enddo

!
! Clean up
!
      deallocate(record)
      ier=fdds_close(fd_in)
      ier=fdds_close(fd_out)

!
! Stop
!
      stop
      end
